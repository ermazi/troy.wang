    如何使用cfssl生成证书

    May 20, 2020

    记录一下cfssl生成证书的方式

    安装cfssl

    curl -L https://github.com/cloudflare/cfssl/releases/download/v1.4.1/cfssl_1.4.1_linux_amd64 -o cfssl
chmod +x cfssl
curl -L https://github.com/cloudflare/cfssl/releases/download/v1.4.1/cfssljson_1.4.1_linux_amd64 -o cfssljson
chmod +x cfssljson
curl -L https://github.com/cloudflare/cfssl/releases/download/v1.4.1/cfssl-certinfo_1.4.1_linux_amd64 -o cfssl-certinfo
chmod +x cfssl-certinfo

    生成证书

    # 全局配置
cfssl print-defaults config > config.json
cfssl print-defaults csr > csr.json

cat > olab-server-csr.json <-EOF
{
  "CN": "kubernetes",
  "hosts": [
    "127.0.0.1",
    "localhost",
    "192.168.199.1",
    "172.17.6.252",
    "kubernetes",
    "kubernetes.default",
    "kubernetes.default.svc",
    "kubernetes.default.svc.cluster",
    "kubernetes.default.svc.cluster.local",
    "*.olab",
    "kubernetes-dashboard"
  ],
  "key": {
    "algo": "rsa",
    "size": 2048
  },
  "names": [{
    "C": "CN",
    "ST": "BJ",
    "L": "BJ",
    "O": "DEVOPS",
    "OU": "OLAB"
  }]
}
EOF

cfssl gencert -initca ca-csr.json | cfssljson -bare ca
cfssl gencert -ca=ca.pem -ca-key=ca-key.pem --config=ca-config.json -profile=kubernetes olab-server-csr.json | cfssljson -bare olab-server

# nginx ingress cert
kubectl create secret tls nginx-cert --key olab-server-key.pem --cert olab-server.pem -n kube-system
